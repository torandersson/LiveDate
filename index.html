LiveDate

A very lightweight "live date" plugin for JQuery.

<div class="liveDate" >

  <span class="show"></span>
    <span style="display:none" class="date">
      Fri Mar 16 2012 10:15:00 GMT+0300 (CET)
    </span>
</div>

<div class="liveDate" >
  <span class="show"></span>
  <span style="display:none" class="date">
    Fri Mar 16 2012 10:17:20 GMT+0400 (CET)
  </span>
</div>


    $(".liveDate").liveDate({ 
        dateSelector: ".date",
        mode: 'countdown',
        callback: function(result){
            $(this).children(".show").html(prettyPrint(result));
        }
    });

    function prettyPrint(result) {
        return result.days + " days, " +  result.hours +" hours, " +  result.minutes +" minutes, " +result.seconds +  " seconds " ;  
    };
