LiveDate

A very lightweight "live date" plugin for JQuery.




 $(".liveDate-DateString").liveDate({ mode: 'countdown', interval: 500, date: function(context) { return new Date($(context).attr("datetime")); }, callback: function(result){ $(this).html(prettyPrint(result)); } }); $(".liveDate-Milliseconds").liveDate({ mode: 'countdown', interval: 1000, date: function(context) { var milliseconds = parseInt($(context).attr("datetime")); return new Date(milliseconds); }, callback: function(result){ $(this).html(prettyPrint(result)); } }); $(".liveDate-NoDate").liveDate({ mode: 'countdown', interval: 1000, callback: function(result){ $(this).html(prettyPrint(result)); } }); function prettyPrint(result) { return result.days + " days, " + result.hours +" hours, " + result.minutes +" minutes, " +result.seconds + " seconds " ; };
